        function formatDate(dateStr) {

            const [year, month, day] = dateStr.split("-");

            const date = new Date(year, month - 1, day);

            return date.toLocaleDateString("en-US", {

                weekday: "short",

                month: "short",

                day: "numeric",

                year: "numeric"

            });

        }


        function buildCard(mover) {

            const isGain = mover.pct_change >= 0;

            const colorClass = isGain ? "gain" : "loss";

            const arrow = isGain ? "▲" : "▼";

            const pctDisplay = `${arrow} ${Math.abs(mover.pct_change).toFixed(2)}%`;

            const summary = mover.ai_summary || "No summary available.";


            return `

        <div class="card ${colorClass}">

          <div class="card-top">

            <div>

              <div class="ticker">${mover.ticker}</div>

              <div class="date">${formatDate(mover.date)}</div>

            </div>

            <div class="pct-badge ${colorClass}">${pctDisplay}</div>

          </div>


          <div class="price-row">

            <span class="close-price">Closing price: <strong>$${mover.close_price.toFixed(2)}</strong></span>

          </div>


          <div class="ai-section">

            <div class="ai-label">✦ AI Summary</div>

            <div class="ai-text">${summary}</div>

          </div>

        </div>

      `;

        }


        function showLoading() {

            document.getElementById("app").innerHTML = `

        <div class="state-container">

          <div class="spinner"></div>

          <p class="state-title">Fetching market data...</p>

        </div>

      `;

        }


        function showError(message) {

            document.getElementById("app").innerHTML = `

        <div class="state-container">

          <div class="error-icon">⚠️</div>

          <p class="state-title">Could not load data</p>

          <p class="error-msg">${message}</p>

        </div>

      `;

        }


        function showEmpty() {

            document.getElementById("app").innerHTML = `

        <div class="state-container">

          <div class="error-icon">📭</div>

          <p class="state-title">No data yet — check back after market close today.</p>

        </div>

      `;

        }


        function showCards(movers) {

            const html = movers.map(buildCard).join("");

            document.getElementById("app").innerHTML = `<div class="grid">${html}</div>`;

        }


        async function loadMovers() {

            showLoading();


            try {

                const response = await fetch(API_URL);


                if (!response.ok) {

                    throw new Error(`API returned status ${response.status}`);

                }


                const data = await response.json();


                if (!data.movers || data.movers.length === 0) {

                    showEmpty();

                    return;

                }


                showCards(data.movers);


            } catch (err) {

                console.error("Failed to fetch movers:", err);

                showError(err.message);

            }

        }


        loadMovers();
